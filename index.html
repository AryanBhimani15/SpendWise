// Privacy-first data handling
async function processStatement(pdfFile) {
// 1. Parse locally in browser
const transactions = await extractData(pdfFile);
// 2. Delete immediately
deleteFile(pdfFile);
// 3. Store ONLY: date, merchant, amount
const safeData = transactions.map(t => ({
date: t.date, // "2024-01-15"
merchant: t.merchant, // "Swiggy"
amount: t.amount, // 450 (INR)
category: t.category // "food_delivery"
}));
// 4. NEVER store: card numbers, names, PDF
return safeData;
}
